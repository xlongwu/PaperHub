## 3. 技术方案设计

### 3.1 数据库 Schema 扩展（Migration v2）

```sql
-- FTS5 全文搜索虚拟表
CREATE VIRTUAL TABLE IF NOT EXISTS document_fts USING fts5(
  title,
  abstract,
  tags,          -- 合并 domain_tags + model_tags 的文本形式
  content='documents',  -- 外部内容表
  content_rowid='rowid'
);

-- 触发器：保持 FTS5 与 documents 表同步
CREATE TRIGGER IF NOT EXISTS documents_fts_insert AFTER INSERT ON documents BEGIN
  INSERT INTO document_fts(rowid, title, abstract, tags)
  VALUES (new.rowid, new.title, new.abstract, 
          json_extract(new.domain_tags, '$') || ' ' || json_extract(new.model_tags, '$'));
END;

CREATE TRIGGER IF NOT EXISTS documents_fts_delete AFTER DELETE ON documents BEGIN
  INSERT INTO document_fts(document_fts, rowid, title, abstract, tags)
  VALUES ('delete', old.rowid, old.title, old.abstract, '');
END;

CREATE TRIGGER IF NOT EXISTS documents_fts_update AFTER UPDATE ON documents BEGIN
  INSERT INTO document_fts(document_fts, rowid, title, abstract, tags)
  VALUES ('delete', old.rowid, old.title, old.abstract, '');
  INSERT INTO document_fts(rowid, title, abstract, tags)
  VALUES (new.rowid, new.title, new.abstract,
          json_extract(new.domain_tags, '$') || ' ' || json_extract(new.model_tags, '$'));
END;

-- 向量索引表（sqlite-vec）
CREATE VIRTUAL TABLE IF NOT EXISTS document_vectors USING vec0(
  document_id TEXT PRIMARY KEY,
  embedding float[1536]  -- OpenAI text-embedding-3-small 维度
);

-- 标签统计表（物化视图，手动维护）
CREATE TABLE IF NOT EXISTS tag_stats (
  tag TEXT PRIMARY KEY,
  category TEXT,        -- 'domain' | 'source' | 'model' | 'type'
  count INTEGER DEFAULT 0,
  last_updated TEXT DEFAULT CURRENT_TIMESTAMP
);
```

**设计说明**：
- FTS5 使用外部内容表模式，避免数据冗余，通过触发器自动同步
- `tags` 列将 JSON 数组扁平化为空格分隔文本，便于 FTS5 索引
- sqlite-vec 向量表独立存储，document_id 作为关联键
- tag_stats 表用于快速渲染标签云，避免实时聚合大量文档

### 3.2 全文搜索模块（FTS5）

```typescript
// src/db/search.ts — FTS5 部分

interface Fts5SearchOptions {
  query: string;           // 用户输入的关键词
  limit?: number;          // 默认 20
  offset?: number;         // 默认 0
  sources?: string[];      // 按来源过滤
  tags?: string[];         // 按标签过滤
}

interface SearchResult {
  document: Document;
  rank: number;            // FTS5 bm25 排名分数
  snippet?: string;        // 高亮片段（可选）
}

export function searchFts5(options: Fts5SearchOptions): SearchResult[] {
  const db = getDb();
  
  // 1. 构建 FTS5 查询字符串
  // 输入: "agent rag" → 输出: "agent* AND rag*"（前缀匹配）
  const ftsQuery = options.query
    .split(/\s+/)
    .filter(Boolean)
    .map(t => `${t}*`)
    .join(' AND ');
  
  // 2. 联合查询 documents + FTS5 排名
  const sql = `
    SELECT d.*, rank FROM document_fts
    JOIN documents d ON d.rowid = document_fts.rowid
    WHERE document_fts MATCH ?
      ${options.sources?.length ? 'AND d.source IN (' + options.sources.map(() => '?').join(',') + ')' : ''}
      ${options.tags?.length ? 'AND (d.domain_tags LIKE ? OR d.model_tags LIKE ?)' : ''}
    ORDER BY rank
    LIMIT ? OFFSET ?
  `;
  
  // 3. 执行并返回
  // ...
}
```

**关键特性**：
- 前缀匹配（`term*`）提升召回率，适合技术术语
- BM25 排名默认排序，可结合时间衰减调整
- 支持来源过滤（arXiv / GPT / Claude）和标签过滤
- 高亮片段：使用 FTS5 `snippet()` 函数提取匹配上下文

```typescript
// src/db/search.ts — 向量搜索部分

interface VectorSearchOptions {
  query: string;           // 用户输入（原始文本，非关键词）
  limit?: number;
  threshold?: number;      // 相似度阈值（默认 0.7）
}

export async function searchVector(options: VectorSearchOptions): Promise<SearchResult[]> {
  // 1. 生成查询文本的 embedding
  const embedding = await generateEmbedding(options.query);
  
  // 2. 使用 sqlite-vec 进行 KNN 搜索
  const sql = `
    SELECT document_id, distance
    FROM document_vectors
    WHERE embedding MATCH ?
      AND k = ?
    ORDER BY distance
  `;
  
  // 3. 转换为 Document 并返回
  // distance 越小越相似，需归一化为 score
}
```

**Embedding 生成策略**：

| 方案 | 实现 | 成本 | 适用场景 |
|------|------|------|---------|
| **OpenAI API** | `text-embedding-3-small`（1536维） | 低（$0.02/1M tokens） | 默认方案，精度高 |
| **本地 ONNX** | `bge-small-en-v1.5`（384维） | 零 API 成本 | 离线环境、成本控制 |

**选择**：Phase 3 优先实现 OpenAI API 方案（简单可靠），本地 ONNX 作为后续优化项。

### 3.4 搜索编排层（混合搜索）

```typescript
// src/search.ts — 搜索编排

export interface HybridSearchOptions {
  query: string;
  mode: 'keyword' | 'semantic' | 'hybrid';  // 默认 hybrid
  limit?: number;
  offset?: number;
  sources?: string[];
  tags?: string[];
  dateRange?: { start: string; end: string };
}

export async function hybridSearch(options: HybridSearchOptions): Promise<{
  results: SearchResult[];
  total: number;
  mode: string;
  report?: string;  // LLM 生成的综述（结果 > 10 时）
}> {
  // 1. 并行执行 FTS5 + 向量搜索
  const [ftsResults, vecResults] = await Promise.all([
    searchFts5({ ...options }),
    options.mode !== 'keyword' ? searchVector({ query: options.query }) : Promise.resolve([]),
  ]);
  
  // 2. 结果聚合与去重
  const merged = mergeResults(ftsResults, vecResults, {
    ftsWeight: 0.6,
    vecWeight: 0.4,
  });
  
  // 3. 应用过滤条件（来源、标签、时间）
  const filtered = applyFilters(merged, options);
  
  // 4. 分页
  const total = filtered.length;
  const page = filtered.slice(offset, offset + limit);
  
  // 5. 若结果 > 10，可选生成 LLM 综述报告
  let report: string | undefined;
  if (page.length > 10 && options.mode === 'hybrid') {
    report = await generateSearchReport(page.slice(0, 10), options.query);
  }
  
  return { results: page, total, mode: options.mode, report };
}
```

**混合排序算法**：

```
final_score = normalize(fts_rank) * 0.6 + normalize(vec_similarity) * 0.4

其中：
- normalize(fts_rank): BM25 分数归一化到 [0, 1]，分数越高越相关
- normalize(vec_similarity): cosine similarity 归一化到 [0, 1]，1 表示完全相同
- 权重可根据搜索结果数量动态调整：
  - 若 FTS 结果 < 5，提升向量权重至 0.6
  - 若向量结果 < 5，提升 FTS 权重至 0.8
```

## 5. 风险与应对

| 风险 | 影响 | 应对措施 |
|------|------|---------|
| **sqlite-vec 编译失败** | 向量搜索不可用 | 准备降级方案：纯 FTS5 搜索；或改用 `sqlite-vss` 替代 |
| **Embedding API 成本** | OpenAI 调用费用累积 | 实现 embedding 缓存（SQLite 表）；同一文档只生成一次；提供本地 ONNX 备选 |
| **FTS5 前缀匹配噪音** | 召回率过高，精度下降 | 提供精确匹配模式（用户加引号 `"term"`）；结果按 BM25 排名过滤低分项 |
| **混合排序效果不佳** | 搜索结果质量差 | 提供 `mode` 参数让用户选择 keyword/semantic/hybrid；A/B 测试不同权重 |
| **数据库迁移失败** | 现有数据丢失 | 迁移前自动备份；使用事务包裹；提供回滚脚本 |
